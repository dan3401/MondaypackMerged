eac_create_governor_ecologist_alpha = {
	create_leader = {
		class = governor
		species = owner_species
		name = random
		skill = 2
		traits = {
			trait = leader_trait_ecologist_alpha
		}
	}
	last_created_leader = { 
		save_event_target_as = eac_last_leader_created
		eac_add_leader_traits_after_creation = yes
	}	
}
