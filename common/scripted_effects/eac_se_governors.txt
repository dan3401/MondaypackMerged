eac_create_governor_bureaucrat = {
	create_leader = {
		class = governor
		species = owner_species
		name = random
		skill = 2
		traits = {
			trait = leader_trait_bureaucrat
			trait = leader_trait_university_educated_governor
		}
	}
	last_created_leader = { 
		save_event_target_as = eac_last_leader_created
		eac_add_leader_traits_after_creation = yes
	}	
}
