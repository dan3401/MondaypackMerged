z_pf_eft_global_cd = {
    if = {
        limit = {
            has_global_flag = z_pf_frequency_high
        }
        set_timed_global_flag = {
            flag = z_pf_cd_glbl_automatic
            days = @z_pf_cd_short
        }
    }
    else_if = {
        limit = {
            has_global_flag = z_pf_frequency_low
        }
        set_timed_global_flag = {
            flag = z_pf_cd_glbl_automatic
            days = @z_pf_cd_long
        }
    }
    else = {
        set_timed_global_flag = {
            flag = z_pf_cd_glbl_automatic
            days = @z_pf_cd_mid
        }
    }
}
