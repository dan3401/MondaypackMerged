namespace = Events_GAPS_Traits_Rowdy

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Rowdy.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Rowdy_Crime" value = 0 }
		Scripted_Planet_Effect_GAPS_Calculate_Rowdy_Crime = yes
		every_owned_pop = {
			remove_modifier = Pop_Modifier_GAPS_Rowdy_Boost
			set_variable = { which = "Pop_Variable_GAPS_Rowdy_Crime" value = 0 }
		}
		while = {
			count = Planet_Variable_GAPS_Rowdy_Crime
			root = {
				every_owned_pop = {
					limit = { pop_has_trait = Trait_GAPS_Rowdy }
					change_variable = { which = "Pop_Variable_GAPS_Rowdy_Crime" value = 1 }
				}
			}
		}
		every_owned_pop = { 
			limit = { pop_has_trait = Trait_GAPS_Rowdy } 
			add_modifier = { 
				modifier = Pop_Modifier_GAPS_Rowdy_Boost 
				days = -1
				multiplier = Pop_Variable_GAPS_Rowdy_Crime
			}
		}
	}
}
#monthly pulse
country_event = {
	id = Events_GAPS_Traits_Rowdy.20
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		any_owned_species = { has_trait = Trait_GAPS_Rowdy }
	}
	immediate = {
		every_owned_planet = { 
			limit = { is_colony = yes }
			planet_event = { id = Events_GAPS_Traits_Rowdy.10 days = 1 }
		}
	}
}
