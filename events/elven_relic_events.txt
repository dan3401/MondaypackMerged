namespace = elven_relic_events

# on_game_start_country event for Origins that start with Relics
# if we end up doing this for any other origins, no need to duplicate, just add more else_ifs
country_event = {
	id = elven_relic_events.0
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_origin = origin_elven_celestial_throne }
			add_relic = r_elven_flaming_pearl
		}
	}
}
