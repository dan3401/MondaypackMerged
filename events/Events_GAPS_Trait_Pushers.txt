namespace = Events_GAPS_Traits_Pushers

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Pushers.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Num_Pushers_Pops" value = 0 }
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Pushers }
			planet = { change_variable = { which = "Planet_Variable_GAPS_Num_Pushers_Pops" value = 1 } }
		}
		remove_modifier = Planet_Modifier_GAPS_Trait_Pushers
		if = {
			limit = { any_owned_pop = { pop_has_trait = Trait_GAPS_Pushers } }
			add_modifier = {
				modifier = Planet_Modifier_GAPS_Trait_Pushers
				days = -1
				multiplier = Planet_Variable_GAPS_Num_Pushers_Pops
			}
		}
	}
}

	}
