namespace = Events_GAPS_Traits_Feral_Tendencies

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Feral_Tendencies.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Num_Feral_Tendencies_Pops" value = 0 }
		every_deposit = {
			limit = { is_deposit_type = Deposits_GAPS_Feral_Tendencies }
			remove_deposit = yes
		}
		remove_modifier = Planet_Modifier_GAPS_Trait_Feral_Tendencies
		remove_modifier = Planet_Modifier_GAPS_Trait_Feral_Tendencies_Gestalt
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Feral_Tendencies }
			planet = { change_variable = { which = "Planet_Variable_GAPS_Num_Feral_Tendencies_Pops" value = 1 } }
		}
		if = {
			limit = { any_owned_pop = { pop_has_trait = Trait_GAPS_Feral_Tendencies } }
			if = { 
				limit = { owner = { is_gestalt = no } }
				add_modifier = { 
					modifier = Planet_Modifier_GAPS_Trait_Feral_Tendencies
					days = -1
					multiplier = Planet_Variable_GAPS_Num_Feral_Tendencies_Pops
				}
			}
			else = {
				add_modifier = { 
					modifier = Planet_Modifier_GAPS_Trait_Feral_Tendencies_Gestalt 
					days = -1
					multiplier = Planet_Variable_GAPS_Num_Feral_Tendencies_Pops
				}
			}
		}
	}
}

	id = Events_GAPS_Traits_Feral_Tendencies.20

	hide_window = yes

		root = { pop_has_trait = Trait_GAPS_Feral_Tendencies }
	}
	immediate = {
		From = {	
			planet_event = { id = Events_GAPS_Traits_Feral_Tendencies.10 days = 1 }
		}
	}
}
