namespace = Events_GAPS_Traits_Nibbler

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Nibbler.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Num_Nibbler_Pops" value = 0 }
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Nibbler }
			planet = { change_variable = { which = "Planet_Variable_GAPS_Num_Nibbler_Pops" value = 1 } }
		}
		remove_modifier = Planet_Modifier_GAPS_Trait_GAPS_Nibbler
		remove_modifier = Planet_Modifier_GAPS_Trait_GAPS_Nibbler_Gestalt
		if = {
			limit = { any_owned_pop = { pop_has_trait = Trait_GAPS_Nibbler } }
			if = { 
				limit = { owner = { is_gestalt = no } }
				add_modifier = { 
					modifier = Planet_Modifier_GAPS_Trait_GAPS_Nibbler
					days = -1
					multiplier = Planet_Variable_GAPS_Num_Nibbler_Pops
				}
			}
			else = {
				add_modifier = { 
					modifier = Planet_Modifier_GAPS_Trait_GAPS_Nibbler_Gestalt 
					days = -1
					multiplier = Planet_Variable_GAPS_Num_Nibbler_Pops
				}
			}
		}
	}
}
