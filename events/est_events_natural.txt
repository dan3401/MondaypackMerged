namespace = est_natural

#Nature 2, find new deposits on a planet.
planet_event = {
	id = est_natural.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			has_tradition = tr_est_natural_2
		}
		num_pops > 0
	}
	
	immediate = {
		if = {
			limit = {
				Not = {
					has_planet_flag = nature_2_flag
					is_planet_class = pc_city
				}
				habitable_structure = no
			}
			set_planet_flag = nature_2_flag
			add_random_non_blocker_deposit = yes
			add_random_non_blocker_deposit = yes
		}
	}
}

#Nature 4 Machines, influence from tile blockers.
planet_event = {
	id = est_natural.2
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		owner = {
			has_swapped_tradition = tr_est_natural_3_machines
		}
		NOT = { has_planet_flag = natural.2.alreadyinfluenced }
	}
	
	immediate = {
		every_deposit = {
			limit = {
				is_blocker = yes
			}
			root = {
				owner = {
					add_monthly_resource_mult = {
						resource = influence
						value = 1
					}
				}
			}
		}
		set_planet_flag = natural.2.alreadyinfluenced
	}
}
