namespace = Events_GAPS_Traits_Delightful

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Delightful.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		#Remove_Old_Delightful_Modifiers = yes
		set_variable = { which = "Planet_Variable_GAPS_Num_Delightful_Pops" value = 0 }
		every_owned_pop = {
			set_variable = { which = "Pop_Variable_GAPS_Num_Delightful_Pops" value = 0 }
		}
		every_owned_pop = {
			remove_modifier = Pop_Modifier_GAPS_Delightful_Bonus
		}
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Delightful }
			root = { change_variable = { which = "Planet_Variable_GAPS_Num_Delightful_Pops" value = 1 } }
		}
		while = {
			count = Planet_Variable_GAPS_Num_Delightful_Pops
			every_owned_pop = { 
				#limit = { NOT = { pop_has_trait = Trait_GAPS_Delightful } }
				change_variable = { which = "Pop_Variable_GAPS_Num_Delightful_Pops" value = 1 }
			}
		}
		every_owned_pop = { 
			limit = { NOT = { pop_has_trait = Trait_GAPS_Delightful } }
			add_modifier = { 
				modifier = Pop_Modifier_GAPS_Delightful_Bonus 
				days = -1
				multiplier = Pop_Variable_GAPS_Num_Delightful_Pops
			}
		}
	}
}

# A pop has been added to the planet
# Root = pop
# From = planet
#on_pop_added
pop_event = {
	id = Events_GAPS_Traits_Delightful.20
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		from = { any_owned_pop = { pop_has_trait = Trait_GAPS_Delightful } }
	}
	immediate = {
		From = {	
			planet_event = { id = Events_GAPS_Traits_Delightful.10 days = 1 }
		}
	}
}
