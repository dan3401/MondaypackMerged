namespace = Events_GAPS_Traits_Builder


#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Builder.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Num_Builder_Pops" value = 0 }
		every_deposit = {
			limit = { is_deposit_type = Deposits_GAPS_Builders }
			remove_deposit = yes
		}
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Builder }
			planet = { change_variable = { which = "Planet_Variable_GAPS_Num_Builder_Pops" value = 1 } }
		}
		remove_modifier = Planet_Modifier_GAPS_Trait_Builders
		if = {
			limit = { any_owned_pop = { pop_has_trait = Trait_GAPS_Builder } }
			add_modifier = {
				modifier = Planet_Modifier_GAPS_Trait_Builders
				days = -1
				multiplier = Planet_Variable_GAPS_Num_Builder_Pops
			}
		}
	}
}
# A pop has been added to the planet
# Root = pop
# From = planet
#on_pop_added
pop_event = {
	id = Events_GAPS_Traits_Builder.20
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		root = { pop_has_trait = Trait_GAPS_Builder }
	}
	immediate = {
		From = {	
			planet_event = { id = Events_GAPS_Traits_Builder.10 days = 1 }
		}
	}
}
