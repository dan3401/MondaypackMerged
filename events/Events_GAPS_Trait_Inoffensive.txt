# A pop has been added to the planet
# Root = pop
# From = planet
#on_pop_added
pop_event = {
	id = Events_GAPS_Traits_Inoffensive.20
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		Root = { pop_has_trait = Trait_GAPS_Inoffensive }
	}
	immediate = {
		From = {	
			planet_event = { id = Events_GAPS_Traits_Inoffensive.10 days = 1 }
		}
	}
}
