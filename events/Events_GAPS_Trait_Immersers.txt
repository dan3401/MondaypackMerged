namespace = Events_GAPS_Traits_Immersers

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Immersers.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Num_Blockers" value = 0 }
		every_deposit = {
			limit = { is_blocker = yes }
			planet = { change_variable = { which = "Planet_Variable_GAPS_Num_Blockers" value = 1 } }
		}
		every_owned_pop = {
			set_variable = { which = "Pop_Variable_GAPS_Num_Blockers" value = 0 }
			remove_modifier = Pop_Modifier_GAPS_Immersers_Boost
			remove_modifier = Pop_Modifier_GAPS_Immersers_Penalty
		}
		if = { 
			limit = { check_variable = { which = "Planet_Variable_GAPS_Num_Blockers" value = 0 } }
			every_owned_pop = {
				limit = { pop_has_trait = Trait_GAPS_Immersers }
				add_modifier = { modifier = Pop_Modifier_GAPS_Immersers_Penalty days = -1 }
			}
		}
		else = { 
			while = {
				count = Planet_Variable_GAPS_Num_Blockers
				every_owned_pop = {
					limit = { pop_has_trait = Trait_GAPS_Immersers }
					change_variable = { which = "Pop_Variable_GAPS_Num_Blockers" value = 1 }
				}
			}
			every_owned_pop = { 
				limit = { pop_has_trait = Trait_GAPS_Immersers } 
				add_modifier = { 
					modifier = Pop_Modifier_GAPS_Immersers_Boost 
					days = -1
					multiplier = Pop_Variable_GAPS_Num_Blockers
				}
			}
		}
		set_variable = { which = "Planet_Variable_GAPS_Num_Immersers_Pops" value = 0 }
		every_deposit = {
			limit = { is_deposit_type = Deposits_GAPS_Immersers }
			remove_deposit = yes
		}
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Immersers }
			planet = { change_variable = { which = "Planet_Variable_GAPS_Num_Immersers_Pops" value = 1 } }
		}
		remove_modifier = Planet_Modifier_GAPS_Trait_Immersers
		if = {
			limit = { any_owned_pop = { pop_has_trait = Trait_GAPS_Immersers } }
			add_modifier = {
				modifier = Planet_Modifier_GAPS_Trait_Immersers
				days = -1
				multiplier = Planet_Variable_GAPS_Num_Immersers_Pops
			}
		}
	}
}
