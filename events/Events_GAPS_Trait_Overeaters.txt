namespace = Events_GAPS_Traits_Overeaters

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Overeaters.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		save_event_target_as = Target_Planet_GAPS_Overeater_Planet
		Scripted_Planet_Effect_Apply_Overeater_Modifiers = yes
		set_variable = { which = "Planet_Variable_GAPS_Num_Overeater_Pops" value = 0 }
		every_owned_pop = {
			remove_modifier = Pop_Modifier_GAPS_Overeaters
			set_variable = { which = "Pop_Variable_GAPS_Num_Overeater_Pops" value = 0 }
		}
		every_owned_pop = {
			limit = { pop_has_trait = Trait_GAPS_Overeaters }
			root = {
				change_variable = { which = "Planet_Variable_GAPS_Num_Overeater_Pops" value = 1 }
			}
		}
		while = {
			count = Planet_Variable_GAPS_Num_Overeater_Pops
			every_owned_pop = {
				limit = { pop_has_trait = Trait_GAPS_Overeaters }
				change_variable = { which = "Pop_Variable_GAPS_Num_Overeater_Pops" value = 1 }
			}
		}
		every_owned_pop = { 
			limit = { pop_has_trait = Trait_GAPS_Overeaters } 
			add_modifier = { 
				modifier = Pop_Modifier_GAPS_Overeaters 
				days = -1
				multiplier = Pop_Variable_GAPS_Num_Overeater_Pops
			}
		}
	}
}
