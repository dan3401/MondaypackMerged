namespace = Events_GAPS_Traits_Sightseers

#CORE EVENT
planet_event = {
	id = Events_GAPS_Traits_Sightseers.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		set_variable = { which = "Planet_Variable_GAPS_Num_Rare_Deposits" value = 0 }
		every_owned_pop = {
			remove_modifier = Pop_Modifier_GAPS_Sightseers_Boost
			set_variable = { which = "Pop_Variable_GAPS_Num_Rare_Deposits" value = 0 }
		}
		every_deposit = {
			limit = { OR = { has_deposit_category = deposit_cat_rare has_deposit_category = deposit_cat_rare_blocker } }
			root = { change_variable = { which = "Planet_Variable_GAPS_Num_Rare_Deposits" value = 1 } }
		}
		Scripted_Planet_Effect_Remove_Old_Sightseers_Modifiers = yes
		while = {
			count = Planet_Variable_GAPS_Num_Rare_Deposits
			root = {
				every_owned_pop = {
					limit = { pop_has_trait = Trait_GAPS_Sightseers }
					change_variable = { which = "Pop_Variable_GAPS_Num_Rare_Deposits" value = 1 }
				}
			}
		}
		every_owned_pop = { 
			limit = { pop_has_trait = Trait_GAPS_Sightseers } 
			add_modifier = { 
				modifier = Pop_Modifier_GAPS_Sightseers_Boost 
				days = -1
				multiplier = Pop_Variable_GAPS_Num_Rare_Deposits
			}
		}
	}
}
# A pop has been added to the planet
# Root = pop
# From = planet

	is_triggered_only = yes
	hide_window = yes

		Root = { pop_has_trait = Trait_GAPS_Sightseers }
	}
	immediate = {
		From = {	
			planet_event = { id = Events_GAPS_Traits_Sightseers.10 days = 1 }
		}
	}
}
